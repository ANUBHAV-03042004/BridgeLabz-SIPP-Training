class AttendanceSystem:
    records: int[students][days]   # 1 = Present, 0 = Absent

    method markAttendance(studentId, day, status)
    method dailySummary(day):
        count = sum(records[allStudents][day])
        return count
    method monthlySummary(studentId):
        return sum(records[studentId][allDays])
    method defaulterList():
        for each student:
            total = monthlySummary(student)
            if total < 0.75 * totalDays:
                print("Student", student, "is a defaulter")

Main:
    system = new AttendanceSystem(30 students, 30 days)
    
    system.markAttendance(1, 1, 1)   # Student 1 present Day 1
    system.markAttendance(3, 1, 0)   # Student 3 absent Day 1

    print(system.dailySummary(10))       # count students present on Day 10
    print(system.monthlySummary(5))      # total present days of Student 5
    system.defaulterList()               # list students < 75%
