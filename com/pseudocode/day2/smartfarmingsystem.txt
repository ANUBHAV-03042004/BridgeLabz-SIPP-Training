abstract class Crop:
    name: String
    abstract method waterRequirement()
    abstract method harvestTime()

class Rice extends Crop:
    method waterRequirement(): return "High water needed"
    method harvestTime(): return "120 days"

class Wheat extends Crop:
    method waterRequirement(): return "Moderate water needed"
    method harvestTime(): return "90 days"

Main:
    crop1 = new Rice()
    crop2 = new Wheat()
    print(crop1.waterRequirement(), crop1.harvestTime())
    print(crop2.waterRequirement(), crop2.harvestTime())
