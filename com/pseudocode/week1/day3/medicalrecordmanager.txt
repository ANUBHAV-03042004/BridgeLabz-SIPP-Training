interface Searchable {
    void search(String query);
}
interface Printable {
    void printReport();
}

class BloodReport implements Searchable, Printable {
    void search(String query) { print("Searching Blood Report: " + query); }
    void printReport() { print("Printing Blood Report..."); }
}

class XRayReport implements Searchable, Printable {
    void search(String query) { print("Searching X-Ray Report: " + query); }
    void printReport() { print("Printing X-Ray Report..."); }
}

main() {
    Printable p1 = new BloodReport();
    Printable p2 = new XRayReport();
    
    p1.printReport();  // Polymorphism
    p2.printReport();
}
