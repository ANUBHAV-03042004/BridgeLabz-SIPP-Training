CLASS TextEditor
    ATTRIBUTE stack

    CONSTRUCTOR()
        INITIALIZE stack as empty

    METHOD type(text)
        PUSH text into stack
        PRINT "Typed: " + text

    METHOD undo()
        IF stack is not empty
            POP lastText from stack
            PRINT "Undo: " + lastText
        ELSE
            PRINT "Nothing to undo"

CLASS Main
    METHOD main()
        CREATE editor = TextEditor()
        CALL editor.type("Hello")
        CALL editor.type("World")
        CALL editor.undo()
