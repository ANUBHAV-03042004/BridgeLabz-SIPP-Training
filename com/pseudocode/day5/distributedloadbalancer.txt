class LoadBalancer:
    servers: Queue<String>

    method addServer(name):
        servers.enqueue(name)

    method removeServer(name):
        temp = empty queue
        while not servers.isEmpty():
            s = servers.dequeue()
            if s != name:
                temp.enqueue(s)
        servers = temp

    method getNextServer():
        s = servers.dequeue()
        servers.enqueue(s)
        return s

Main:
    lb = new LoadBalancer()
    lb.addServer("S1")
    lb.addServer("S2")
    lb.addServer("S3")

    print("R1 -> " + lb.getNextServer())   # S1
    print("R2 -> " + lb.getNextServer())   # S2
    print("R3 -> " + lb.getNextServer())   # S3

    lb.addServer("S4")
    print("R4 -> " + lb.getNextServer())   # S1

    lb.removeServer("S2")
    print("R5 -> " + lb.getNextServer())   # S3 (S2 skipped)
