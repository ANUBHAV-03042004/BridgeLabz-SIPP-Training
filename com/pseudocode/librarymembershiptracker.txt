CLASS Member
    ATTRIBUTE name
    ATTRIBUTE memberId
    ATTRIBUTE membershipDate

    CONSTRUCTOR(name, memberId, membershipDate)
        SET this.name = name
        SET this.memberId = memberId
        SET this.membershipDate = membershipDate

    METHOD checkValidity(currentDate)
        IF currentDate - membershipDate <= 365
            PRINT "Membership Valid"
        ELSE
            PRINT "Membership Expired"

CLASS Main
    METHOD main()
        CREATE member1 = Member("John", "M101", "2024-01-15")
        CALL member1.checkValidity("2025-01-10")
