CLASS Employee
    ATTRIBUTE name
    ATTRIBUTE department

    CONSTRUCTOR(name, department)
        SET this.name = name
        SET this.department = department

CLASS Main
    METHOD main()
        CREATE employees = ArrayList<Employee>
        ADD Employee("Ravi", "IT") into employees
        ADD Employee("Sita", "HR") into employees
        ADD Employee("Amit", "IT") into employees

        FILTER employees where department == "IT"
        PRINT filtered employees
