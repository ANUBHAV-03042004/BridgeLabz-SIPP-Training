class Doctor:
    name: String
    specialization: String

Map<String, List<Doctor>> hospitalMap

hospitalMap["Cardiology"] = [Doctor("A", "Heart"), Doctor("B", "Heart")]
hospitalMap["Neurology"] = [Doctor("C", "Brain")]

method filterDoctorsBySpecialization(specialization):
    for each department in hospitalMap:
        for doctor in hospitalMap[department]:
            if doctor.specialization == specialization:
                print doctor.name, "in", department
