class WordCounter:
    method countWords(paragraph):
        cleanText = paragraph.toLowerCase().removePunctuation()
        words = split(cleanText, " ")
        freqMap = Map<String, Integer>
        
        for word in words:
            freqMap[word] = freqMap.getOrDefault(word, 0) + 1

        sortedMap = sortByValue(freqMap, descending)
        return sortedMap

Main:
    text = "The news today is about the rise in technology trends."
    wordCounter = new WordCounter()
    result = wordCounter.countWords(text)
    print(result)   # e.g., the:2, news:1, today:1 ...
