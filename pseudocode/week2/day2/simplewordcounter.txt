class WordCounter:
    method countWords(paragraph):
        words = paragraph.toLowerCase().removePunctuation().split(" ")
        freqMap = Map<String, Integer>
        
        for word in words:
            freqMap[word] = freqMap.getOrDefault(word, 0) + 1

        sortedMap = sortByValue(freqMap, descending)
        return sortedMap

Main:
    text = "Sports bring people together. Sports inspire teamwork. Sports create leaders."
    counter = new WordCounter()
    print(counter.countWords(text))
    # Output: sports=3, bring=1, people=1, ...
