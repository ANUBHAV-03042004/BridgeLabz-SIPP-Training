// Class ClinicManager
class ClinicManager:
    pets = []

    function registerPet(pet):
        pets.add(pet)
        print("Registering: " + pet.name + " (" + pet.species + ")")
        status = pet.checkVaccinationStatus()
        print("Status: " + status)
