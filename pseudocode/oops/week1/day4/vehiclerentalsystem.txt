interface Rentable:
    method rent(days: int): double

abstract class Vehicle implements Rentable:
    type: String
    basePrice: double

    constructor(type, basePrice):
        this.type = type
        this.basePrice = basePrice

class Car extends Vehicle:
    method rent(days: int):
        return basePrice * days

class Bike extends Vehicle:
    method rent(days: int):
        return (basePrice * days) * 0.8   # Discount for bikes

class VehicleFactory:
    method createVehicle(type: String):
        if type == "Car": return new Car("Car", 1000)
        if type == "Bike": return new Bike("Bike", 500)

Main:
    factory = new VehicleFactory()
    v1 = factory.createVehicle("Car")
    v2 = factory.createVehicle("Bike")

    print(v1.rent(3))   # Car rent
    print(v2.rent(3))   # Bike rent
