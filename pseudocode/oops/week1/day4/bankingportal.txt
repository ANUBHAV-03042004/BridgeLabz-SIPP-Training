class Account:
    accountNumber: int
    balance: double

    method deposit(amount: double):
        balance = balance + amount

    method withdraw(amount: double):   # base behavior
        if balance >= amount:
            balance = balance - amount
        else:
            print("Insufficient Balance")

class Saving extends Account:
    method withdraw(amount: double):
        if balance - amount >= 500:   # Min balance rule
            balance = balance - amount
        else:
            print("Cannot withdraw. Minimum balance required: 500")

class Current extends Account:
    overdraftLimit: double = 10000

    method withdraw(amount: double):
        if balance + overdraftLimit >= amount:
            balance = balance - amount
        else:
            print("Overdraft limit exceeded")

Main:
    acc1 = new Saving()
    acc1.deposit(2000)
    acc1.withdraw(1800)   # Should fail

    acc2 = new Current()
    acc2.deposit(1000)
    acc2.withdraw(5000)   # Allowed due to overdraft
