CLASS Pet 
	ATTRIBUTE name
	ATTRIBUTE age
	ATTRIBUTE breed
	
	CONSTRUCTOR(name, age, breed)
		SET this.name = name
		SET this.age = age
		SET this.breed = breed
		
	METHOD displayInfo()
		PRINT "Name " + this.name
		PRINT "Age " + this.age
		PRINT "Breed " + this.breed
		
CLASS Dog EXTENDS Pet
	ATTRIBUTE isWalk
	
	CONSTRUCTOR(name,age,breed,isWalk)
		CALL super(name,age,breed)
		SET this.isWalk = isWalk
		
	METHOD displayInfo()
		CALL super.displayInfo()
		PRINT "Is Walk " + isWalk
		
	
CLASS Cat EXTENDS Pet
	ATTRIBUTE isIndoor
	
	CONSTRUCTOR(name,age,breed,isIndoor)
		CALL super(name,age,breed)
		SET this.isIndoor = isIndoor
		
	METHOD displayInfo()
		CALL super.displayInfo()
		PRINT "Is Indoor " + isIndoor
	
	
CLASS Bird EXTENDS Pet
	ATTRIBUTE canTalk
	
	CONSTRUCTOR(name,age,breed,canTalk)
		CALL super(name,age,breed)
		SET this.canTalk = canTalk
