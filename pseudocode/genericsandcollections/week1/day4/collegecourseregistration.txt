class Student:
    name: String

class Course:
    courseName: String

Main:
    courseMap = Map<Course, List<Student>>()

    c1 = new Course("Math")
    c2 = new Course("CS")

    s1 = new Student("Alice")
    s2 = new Student("Bob")
    s3 = new Student("Charlie")

    courseMap[c1] = [s1, s2]
    courseMap[c2] = [s3]

    for course in courseMap:
        print(course.courseName, ":", courseMap[course])
