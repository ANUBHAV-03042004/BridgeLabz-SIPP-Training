class Employee:
    name: String
    dept: String
    rating: int
    salary: double

Main:
    employees = [
        Employee("A", "IT", 5, 50000),
        Employee("B", "IT", 3, 30000),
        Employee("C", "IT", 2, 20000),
        Employee("D", "Finance", 4, 40000)
    ]

    # Group employees by department
    grouped = groupBy(employees, e -> e.dept)

    for dept in grouped:
        empList = grouped[dept]

        totalSalary = sum(e.salary for e in empList)
        pool = totalSalary * 0.10

        totalRating = sum(e.rating for e in empList)

        for e in empList:
            share = (e.rating / totalRating) * pool
            print(e.name, dept, "Bonus:", share)
