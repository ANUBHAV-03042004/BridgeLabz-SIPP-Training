class TaskOrganizer:
    tasks: LinkedHashMap<Integer, String>
    taskCounter: int

    method addTask(desc):
        taskCounter++
        tasks.put(taskCounter, desc)

    method removeTask(taskId):
        tasks.remove(taskId)

    method viewTasks():
        for (id, desc) in tasks:
            print(id + " → " + desc)

Main:
    organizer = new TaskOrganizer()
    organizer.addTask("Check emails")
    organizer.addTask("Prepare sales report")
    organizer.addTask("Call client")

    organizer.removeTask(2)
    organizer.viewTasks()
