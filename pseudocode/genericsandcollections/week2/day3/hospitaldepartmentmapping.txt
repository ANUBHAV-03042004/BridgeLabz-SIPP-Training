class Doctor:
    name: String
    specialization: String
    experience: int

Main:
    hospital = Map<String, List<Doctor>>()
    hospital["Cardiology"] = [ new Doctor("Dr. Mehta","Cardiologist",15), new Doctor("Dr. Kapoor","Cardiologist",10) ]
    hospital["Neurology"] = [ new Doctor("Dr. Sharma","Neurologist",10), new Doctor("Dr. Das","Neurologist",8) ]
    hospital["Pediatrics"] = [ new Doctor("Dr. Ritu","Pediatrician",12), new Doctor("Dr. Anjali","Pediatrician",7) ]

    query = "Neurology"
    print("Doctors in Neurology:")
    for d in hospital[query]:
        print(d.name + " - " + d.specialization + " (" + d.experience + " years)")
