class Transaction<T> {
    T paymentMethod;
    double amount;

    Transaction(T paymentMethod, double amount) {
        this.paymentMethod = paymentMethod;
        this.amount = amount;
    }

    void process() {
        print("Processing " + paymentMethod + " with amount " + amount);
    }
}

main() {
    Transaction<String> upi = new Transaction<>("UPI", 500);
    Transaction<String> card = new Transaction<>("Card", 1500);

    upi.process();
    card.process();
}
