boolean solveMaze(int maze[][], int x, int y, int sol[][]) {
    if (x == N-1 && y == N-1) { 
        sol[x][y] = 1; return true; 
    }

    if (isSafe(maze, x, y)) {
        sol[x][y] = 1;
        if (solveMaze(maze, x+1, y, sol)) return true;
        if (solveMaze(maze, x, y+1, sol)) return true;
        sol[x][y] = 0; // backtrack
        return false;
    }
    return false;
}
